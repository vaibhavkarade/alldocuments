# Variables for resource group and factory names
resourceGroup="<Prod-Resource-Group-Name>"
factoryName="<Prod-ADF-Name>"
devResourceGroup="<Dev-Resource-Group-Name>"
devFactoryName="<Dev-ADF-Name>"

# ---- Pipelines ----

# Get list of pipelines in prod and dev
prodPipelines=$(az datafactory pipeline list --resource-group $resourceGroup --factory-name $factoryName --query [].name --output tsv)
devPipelines=$(az datafactory pipeline list --resource-group $devResourceGroup --factory-name $devFactoryName --query [].name --output tsv)

# Compare and delete pipelines that exist in prod but not in dev
for pipeline in $prodPipelines; do
    if [[ ! " $devPipelines " =~ " $pipeline " ]]; then
        echo "Deleting extra pipeline: $pipeline"
        az datafactory pipeline delete --resource-group $resourceGroup --factory-name $factoryName --name $pipeline --yes
    fi
done
