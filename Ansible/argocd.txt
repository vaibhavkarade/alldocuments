---
- name: Install ArgoCD
  hosts: all
  become: yes

  tasks:
    - name: Download ArgoCD CLI binary
      get_url:
        url: "https://github.com/argoproj/argo-cd/releases/latest/download/argocd-linux-amd64"
        dest: /usr/local/bin/argocd
        mode: 'a+x'

    - name: Create systemd service unit file for ArgoCD
      ansible.builtin.copy:
        dest: /etc/systemd/system/argocd.service
        content: |
          [Unit]
          Description=ArgoCD Server
          After=network.target

          [Service]
          Type=simple
          ExecStart=/usr/local/bin/argocd server --insecure

          [Install]
          WantedBy=multi-user.target

    - name: Reload systemd and start ArgoCD service
      ansible.builtin.systemd:
        name: argocd
        state: started
        enabled: yes
