---
- name: Install Terraform
  hosts: all
  become: yes  # Use elevated privileges

  tasks:
    - name: Ensure unzip is installed
      package:
        name: unzip
        state: present

    - name: Create directory for Terraform
      file:
        path: /usr/local/bin
        state: directory

    - name: Download Terraform binary
      get_url:
        url: "https://releases.hashicorp.com/terraform/1.1.0/terraform_1.1.0_linux_amd64.zip"
        dest: /tmp/terraform.zip

    - name: Extract Terraform binary
      unarchive:
        src: /tmp/terraform.zip
        dest: /usr/local/bin
        remote_src: yes

    - name: Set permissions for Terraform binary
      file:
        path: /usr/local/bin/terraform
        mode: +x
