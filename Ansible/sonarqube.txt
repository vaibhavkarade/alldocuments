---
- name: Install SonarQube container
  hosts: all
  become: yes
  tasks:
    - name: Pull SonarQube Docker image
      docker_image:
        name: sonarqube
        source: pull
        state: present

    - name: Run SonarQube container
      docker_container:
        name: sonarqube
        image: sonarqube:latest
        ports:
          - "9000:9000"
          - "9092:9092"
        volumes:
          - "/path/to/sonarqube_data:/opt/sonarqube/data"
          - "/path/to/sonarqube_extensions:/opt/sonarqube/extensions"
        restart_policy: always
