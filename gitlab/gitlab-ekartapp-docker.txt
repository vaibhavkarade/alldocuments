stages:
  - build
  - dockerize
  - deploy

variables:
  DOCKER_IMAGE_NAME: "ekart-gitlab"
  DOCKER_REGISTRY: "docker.io"
  DOCKER_IMAGE_TAG: "latest"
  DOCKER_USERNAME: "vaibhavkarade"
  DOCKER_PASSWORD: "Bunty@12345*"

build:
  stage: build
  image: maven:3.8.3-openjdk-11
  script:
    - mvn clean install -DskipTests=true
  artifacts:
      paths:
          - target/shopping-cart-0.0.1-SNAPSHOT.jar



dockerize:
  stage: dockerize
  image: docker:19.03.12
  services:
    - docker:19.03.12-dind
  before_script:
    - echo "$DOCKER_PASSWORD" | docker login -u "$DOCKER_USERNAME" --password-stdin $DOCKER_REGISTRY
  script:
    - docker build -t $DOCKER_REGISTRY/$DOCKER_USERNAME/$DOCKER_IMAGE_NAME:$DOCKER_IMAGE_TAG .
    - docker push $DOCKER_REGISTRY/$DOCKER_USERNAME/$DOCKER_IMAGE_NAME:$DOCKER_IMAGE_TAG

deploy:
  stage: deploy
  image: docker:19.03.12
  services:
    - docker:19.03.12-dind
  before_script:
    - echo "$DOCKER_PASSWORD" | docker login -u "$DOCKER_USERNAME" --password-stdin $DOCKER_REGISTRY
  script:
    - docker run -d --name $DOCKER_IMAGE_NAME -p 8050:8050 $DOCKER_REGISTRY/$DOCKER_USERNAME/$DOCKER_IMAGE_NAME:$DOCKER_IMAGE_TAG
