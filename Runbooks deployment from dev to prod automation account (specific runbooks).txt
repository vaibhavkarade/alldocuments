trigger: none
pr: none

variables:
  devSubscriptionId: '8172f712-63ee-44d9-8b59-3fa257b9dcbd'
  prodSubscriptionId: '8172f712-63ee-44d9-8b59-3fa257b9dcbd'
  devResourceGroup: 'cicd-auto'
  devAutomationAccount: 'devautocicd'
  prodResourceGroup: 'cicd-auto'
  prodAutomationAccount: 'prodautocicd'
  runbookNames: 'runbook1,runbook2'           # blank = all runbooks
  recreateRunbooks: 'false'                   # set to 'true' to delete before import

stages:

# ---------------------------------------------------------------------------
- stage: ExportRunbooks
  displayName: 'Export selected runbooks from Dev'
  jobs:
  - job: Export
    displayName: 'Export Dev runbooks'
    pool:
      vmImage: 'windows-latest'
    steps:
    - task: AzurePowerShell@5
      displayName: 'Export runbooks'
      inputs:
        azureSubscription: 'svc-conn-dev-automation'
        ScriptType: InlineScript
        azurePowerShellVersion: LatestVersion
        pwsh: true
        Inline: |
          Install-Module Az.Automation -Scope CurrentUser -Force -AllowClobber

          $devSubscriptionId    = "$(devSubscriptionId)"
          $devResourceGroup     = "$(devResourceGroup)"
          $devAutomationAccount = "$(devAutomationAccount)"
          $requestedNamesText   = "$(runbookNames)"

          Write-Host "Setting context to Dev subscription $devSubscriptionId"
          Set-AzContext -SubscriptionId $devSubscriptionId | Out-Null

          $outputFolder = "$(Build.ArtifactStagingDirectory)"
          New-Item -Path $outputFolder -ItemType Directory -Force | Out-Null

          Write-Host "Retrieving runbooks from Dev Automation account ..."
          $runbooks = Get-AzAutomationRunbook `
                        -ResourceGroupName $devResourceGroup `
                        -AutomationAccountName $devAutomationAccount `
                        -ErrorAction Stop

          if (-not $runbooks) {
            Write-Warning "No runbooks found in Dev automation account."
          }

          $requestedNames = @()
          if (-not [string]::IsNullOrWhiteSpace($requestedNamesText)) {
              $requestedNames = $requestedNamesText `
                                -split "[,\r\n]" `
                                | ForEach-Object { $_.Trim() } `
                                | Where-Object { $_ }

              if ($requestedNames.Count -gt 0) {
                  $runbooks = $runbooks | Where-Object { $requestedNames -contains $_.Name }

                  foreach ($name in $requestedNames) {
                      if (-not ($runbooks.Name -contains $name)) {
                          Write-Warning "Requested runbook '$name' was not found in Dev and will be skipped."
                      }
                  }
              }
          }

          if (-not $runbooks) {
              Write-Warning "No runbooks match the selection criteria. Nothing to export."
          }

          $metadata = @()

          foreach ($runbook in $runbooks) {
            Write-Host ("Exporting runbook '{0}' (Type = {1}, RuntimeVersion = {2})..." `
                        -f $runbook.Name, $runbook.RunbookType, $runbook.RuntimeVersion)

            Export-AzAutomationRunbook `
              -ResourceGroupName $devResourceGroup `
              -AutomationAccountName $devAutomationAccount `
              -Name $runbook.Name `
              -Slot Published `
              -OutputFolder $outputFolder `
              -Force | Out-Null

            $exportedFile = Get-ChildItem -Path $outputFolder -Filter "$($runbook.Name).*" -File |
                            Sort-Object LastWriteTime -Descending |
                            Select-Object -First 1

            if (-not $exportedFile) {
              throw "Failed to locate exported file for runbook '$($runbook.Name)'."
            }

            $metadata += [pscustomobject]@{
              Name           = $runbook.Name
              RunbookType    = $runbook.RunbookType
              RuntimeVersion = $runbook.RuntimeVersion
              File           = $exportedFile.Name
            }
          }

          $metadataPath = Join-Path $outputFolder 'runbooks.json'
          $metadata | ConvertTo-Json -Depth 5 | Out-File -FilePath $metadataPath -Encoding UTF8

          Write-Host "Export complete. Metadata written to $metadataPath (Runbooks exported: $($metadata.Count))."
    - publish: $(Build.ArtifactStagingDirectory)
      artifact: RunbookPackage

# ---------------------------------------------------------------------------
- stage: ImportRunbooks
  displayName: 'Import runbooks into Prod'
  dependsOn: ExportRunbooks
  jobs:
  - job: Import
    displayName: 'Import to Prod'
    pool:
      vmImage: 'windows-latest'
    steps:
    - download: current
      artifact: RunbookPackage
    - task: AzurePowerShell@5
      displayName: 'Import & publish runbooks'
      inputs:
        azureSubscription: 'svc-conn-dev-automation'
        ScriptType: InlineScript
        azurePowerShellVersion: LatestVersion
        pwsh: true
        Inline: |
          Install-Module Az.Automation -Scope CurrentUser -Force -AllowClobber

          $prodSubscriptionId    = "$(prodSubscriptionId)"
          $prodResourceGroup     = "$(prodResourceGroup)"
          $prodAutomationAccount = "$(prodAutomationAccount)"
          $recreateAll           = [System.Convert]::ToBoolean("$(recreateRunbooks)")

          Write-Host "Setting context to Prod subscription $prodSubscriptionId"
          Set-AzContext -SubscriptionId $prodSubscriptionId | Out-Null

          $artifactFolder = "$(Pipeline.Workspace)/RunbookPackage"
          $metadataPath   = Join-Path $artifactFolder 'runbooks.json'

          if (-not (Test-Path $metadataPath)) {
            throw "Metadata file 'runbooks.json' not found in artifact. Cannot continue."
          }

          $metadata = Get-Content -Path $metadataPath -Raw | ConvertFrom-Json
          if ($metadata -isnot [System.Collections.IEnumerable]) {
            $metadata = @($metadata)
          }

          $typeMap = [System.Collections.Generic.Dictionary[string,string]]::new([System.StringComparer]::OrdinalIgnoreCase)
          $typeMap["PowerShell"]                  = "PowerShell"
          $typeMap["PowerShellWorkflow"]          = "PowerShellWorkflow"
          $typeMap["Graph"]                       = "Graph"
          $typeMap["GraphPowerShell"]             = "GraphicalPowerShell"
          $typeMap["GraphicalPowerShell"]         = "GraphicalPowerShell"
          $typeMap["GraphPowerShellWorkflow"]     = "GraphicalPowerShellWorkflow"
          $typeMap["GraphicalPowerShellWorkflow"] = "GraphicalPowerShellWorkflow"
          $typeMap["Python2"]                     = "Python2"
          $typeMap["Python3"]                     = "Python3"
          $typeMap["Powershell72"]                = "PowerShell72"
          $typeMap["PowerShell7"]                 = "PowerShell72"

          function Get-CanonicalRunbookType {
              param([string]$TypeName)
              if ([string]::IsNullOrWhiteSpace($TypeName)) { return $null }
              if ($typeMap.ContainsKey($TypeName)) { return $typeMap[$TypeName] }
              return $TypeName
          }

          $importParameters = (Get-Command -Name Import-AzAutomationRunbook -ErrorAction Stop).Parameters
          $runtimeParamSupported = $importParameters.ContainsKey("RuntimeVersion")

          foreach ($item in $metadata) {
            $runbookFile = Join-Path $artifactFolder $item.File

            if (-not (Test-Path $runbookFile)) {
              throw "Runbook file '$($item.File)' referenced in metadata not found."
            }

            $importType    = Get-CanonicalRunbookType $item.RunbookType
            $importRuntime = $item.RuntimeVersion

            if (-not $importType) {
              Write-Warning ("Runbook '{0}' has null/empty type value. Skipping." -f $item.Name)
              continue
            }

            $deleteBeforeImport = $recreateAll

            $existingRunbook = Get-AzAutomationRunbook `
                                  -ResourceGroupName $prodResourceGroup `
                                  -AutomationAccountName $prodAutomationAccount `
                                  -Name $item.Name `
                                  -ErrorAction SilentlyContinue

            if ($existingRunbook) {
              $existingType    = Get-CanonicalRunbookType $existingRunbook.RunbookType
              $existingRuntime = $existingRunbook.RuntimeVersion

              Write-Host ("Existing runbook '{0}' found in Prod. Type = {1}, RuntimeVersion = {2}" `
                          -f $item.Name, $existingType, $existingRuntime)

              if (-not $deleteBeforeImport) {
                if ($existingType -ne $importType) {
                    Write-Warning ("Type mismatch for '{0}' (Prod = {1}, Dev = {2}). It will be recreated." `
                                   -f $item.Name, $existingType, $importType)
                    $deleteBeforeImport = $true
                }
                elseif ($existingRuntime -ne $importRuntime) {
                    Write-Warning ("Runtime mismatch for '{0}' (Prod = {1}, Dev = {2}). It will be recreated." `
                                   -f $item.Name, $existingRuntime, $importRuntime)
                    $deleteBeforeImport = $true
                }
              }

              if ($deleteBeforeImport) {
                Write-Host ("Removing existing runbook '{0}' in Prod prior to import." -f $item.Name)
                Remove-AzAutomationRunbook `
                  -ResourceGroupName $prodResourceGroup `
                  -AutomationAccountName $prodAutomationAccount `
                  -Name $item.Name `
                  -Force `
                  -Confirm:$false `
                  | Out-Null
                $existingRunbook = $null
              }
            }

            Write-Host ("Importing runbook '{0}' (Dev Type = {1}, RuntimeVersion = {2}) from '{3}'..." `
                        -f $item.Name, $importType, $importRuntime, $item.File)

            $importCmd = @{
              ResourceGroupName     = $prodResourceGroup
              AutomationAccountName = $prodAutomationAccount
              Name                  = $item.Name
              Path                  = $runbookFile
              Type                  = $importType
              Force                 = $true
            }
            if ($runtimeParamSupported -and $importRuntime) {
                $importCmd["RuntimeVersion"] = $importRuntime
            }

            Import-AzAutomationRunbook @importCmd | Out-Null

            Publish-AzAutomationRunbook `
              -ResourceGroupName $prodResourceGroup `
              -AutomationAccountName $prodAutomationAccount `
              -Name $item.Name `
              | Out-Null
          }

          Write-Host "Import and publish of selected runbooks completed successfully."
