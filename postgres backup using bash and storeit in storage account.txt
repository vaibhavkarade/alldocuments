#!/bin/bash

# Variables
PG_DB="mydatabase"
PG_USER="postgres"
BACKUP_FILE="/tmp/pg_backup_$(date +'%Y%m%d%H%M%S').sql"
STORAGE_ACCOUNT="mypgstorageacct"
CONTAINER="pgbackups"
ACCOUNT_KEY="your_storage_account_key"

echo "=== Starting PostgreSQL Backup ==="

# Step 1: Backup PostgreSQL Database
sudo -u $PG_USER pg_dump $PG_DB > $BACKUP_FILE

if [ $? -eq 0 ]; then
    echo "✔ Backup successful: $BACKUP_FILE"
else
    echo "❌ Backup failed!"
    exit 1
fi

# Step 2: Upload Backup to Azure Storage
az storage blob upload \
  --account-name $STORAGE_ACCOUNT \
  --container-name $CONTAINER \
  --name $(basename $BACKUP_FILE) \
  --file $BACKUP_FILE \
  --account-key $ACCOUNT_KEY

if [ $? -eq 0 ]; then
    echo "✔ Backup uploaded successfully."
else
    echo "❌ Upload failed!"
    exit 1
fi

echo "🎉 Backup Process Completed!"
