# ----------------------- CONFIG -----------------------
$ApplicationId         = "0c8a1f70-c455-4c73-8df9-5ed54cb0f942"  # SPN Application (client) ID
$SecretName            = "spnsecretrot12345678"                        # Secret name in Key Vault
$KeyVaultName          = "testspnkeyvault"                       # Key Vault name
$ThresholdDays         = 15                                      # Rotation threshold in days
$SecretValidityInYears = 1                                       # Validity of new secret
# -------------------------------------------------------

$ErrorActionPreference = "Stop"

Write-Output "🔐 Connecting to Azure..."
Connect-AzAccount -Identity

# Get the SPN object
Write-Output "🔍 Fetching SPN details..."
$spn = Get-AzADApplication -ApplicationId $ApplicationId
if (-not $spn) {
    throw "❌ Application with AppId $ApplicationId not found."
}

# Get all existing secrets for the app
$existingSecrets = Get-AzADAppCredential -ApplicationId $ApplicationId
$rotate = $false
$oldSecretId = $null

if ($existingSecrets) {
    foreach ($cred in $existingSecrets) {
        $expiry = $cred.EndDateTime
        if ($expiry -lt (Get-Date).AddDays($ThresholdDays)) {
            Write-Output "⚠️ Credential expiring on $expiry. Rotation needed."
            $rotate = $true
            $oldSecretId = $cred.KeyId
            break
        } else {
            Write-Output "✅ Secret valid till $expiry"
        }
    }
} else {
    Write-Output "ℹ️ No credentials found. Will create a new one."
    $rotate = $true
}

if ($rotate) {
    $startDate = Get-Date
    $endDate   = $startDate.AddYears($SecretValidityInYears)
    $displayName = "AutoRotated-$(Get-Date -Format 'yyyyMMddHHmmss')"

    Write-Output "🔧 Creating new SPN secret..."
    $passwordCred = @{
        DisplayName   = $displayName
        StartDateTime = $startDate
        EndDateTime   = $endDate
    }

    $newCred = New-AzADAppCredential -ApplicationId $ApplicationId `
                                     -PasswordCredential @($passwordCred)

    $newPassword = $newCred.SecretText
    if ([string]::IsNullOrWhiteSpace($newPassword)) {
        throw "❌ No secretText returned by New-AzADAppCredential."
    }

    Write-Output "🔐 Storing new secret in Key Vault..."
    $secureSecret = ConvertTo-SecureString $newPassword -AsPlainText -Force
    $newKvSecret = Set-AzKeyVaultSecret -VaultName $KeyVaultName `
                                        -Name $SecretName `
                                        -SecretValue $secureSecret `
                                        -Expires $endDate
    Write-Output "✅ Stored new version of secret in Key Vault."

    # Disable old Key Vault versions
    Write-Output "🔒 Disabling older Key Vault versions..."
    $currentVersion = $newKvSecret.Version
    $allVersions = Get-AzKeyVaultSecret -VaultName $KeyVaultName -Name $SecretName -IncludeVersions

    foreach ($version in $allVersions) {
        $versionId = $version.Id.Split("/")[-1]
        $enabledFlag = if ($null -eq $version.Attributes.Enabled) { $true } else { $version.Attributes.Enabled }

        if ($versionId -ne $currentVersion -and $enabledFlag) {
            Update-AzKeyVaultSecret -VaultName $KeyVaultName `
                                    -Name $SecretName `
                                    -Version $versionId `
                                    -Enable $false | Out-Null
            Write-Output "⛔ Disabled old Key Vault version: $versionId"
        }
    }

    # Delete old SPN secret
    if ($oldSecretId) {
        Write-Output "🗑️ Deleting old SPN secret with KeyId: $oldSecretId"
        Remove-AzADAppCredential -ApplicationId $ApplicationId -KeyId $oldSecretId
        Write-Output "✅ Old SPN secret deleted."
    }

    Write-Output "✅ Secret rotation complete."
} else {
    Write-Output "👍 All secrets are valid for more than $ThresholdDays days. No rotation needed."
}
