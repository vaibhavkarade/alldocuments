# MAGIC ### 🔍 Find Inactive Repositories (Checking ALL Branches)

# COMMAND ----------

# Setup API authentication
authorization = str(b64encode(bytes(':' + pat, 'ascii')), 'ascii')
headers = {
    'Authorization': f'Basic {authorization}',
    'Content-Type': 'application/json'
}

threshold_date = datetime.now() - timedelta(days=days_inactive)

# Get all repositories
repos_url = f"https://dev.azure.com/{organization}/{project}/_apis/git/repositories?api-version=7.0"

try:
    response = requests.get(repos_url, headers=headers, timeout=30)
    response.raise_for_status()
    repositories = response.json()['value']
    print(f"📦 Found {len(repositories)} total repositories\n")
except Exception as e:
    print(f"❌ Error: {e}")
    dbutils.notebook.exit("Failed to fetch repositories")

# Find inactive ones
inactive_repos = []

print(f"🔍 Scanning repositories {'across ALL branches' if check_all_branches else 'on default branch only'}...\n")

for idx, repo in enumerate(repositories, 1):
    repo_name = repo['name']
    repo_id = repo['id']
    
    if idx % 25 == 0:
        print(f"   Processed {idx}/{len(repositories)} repositories...")
    
    try:
        most_recent_commit = None
        most_recent_date = None
        most_active_branch = None
        all_recent_committers = set()
        all_recent_emails = set()
        active_branches_list = []
        
        if check_all_branches:
            # 🔴 NEW: Get ALL branches first
            branches_url = f"https://dev.azure.com/{organization}/{project}/_apis/git/repositories/{repo_id}/refs?filter=heads/&api-version=7.0"
            
            try:
                branches_response = requests.get(branches_url, headers=headers, timeout=30)
                branches_response.raise_for_status()
                branches_data = branches_response.json()
                
                if branches_data.get('count', 0) == 0:
                    # No branches, treat as empty
                    inactive_repos.append({
                        'Repository_Name': repo_name,
                        'Last_Commit_Date': 'Never (Empty)',
                        'Days_Inactive': 'N/A',
                        'Most_Active_Branch': 'N/A',
                        'Active_Branches_Count': 0,
                        'Last_Committer_Name': 'N/A',
                        'Last_Committer_Email': 'N/A',
                        'All_Recent_Committers': 'N/A',
                        'Commit_Message': 'N/A',
                        'Web_URL': repo['webUrl']
                    })
                    continue
                
                # 🔴 NEW: Check each branch for activity
                for branch_ref in branches_data.get('value', []):
                    branch_name = branch_ref['name'].replace('refs/heads/', '')
                    
                    # Get commits for this specific branch
                    commits_url = f"https://dev.azure.com/{organization}/{project}/_apis/git/repositories/{repo_id}/commits?searchCriteria.itemVersion.version={branch_name}&$top=5&api-version=7.0"
                    
                    try:
                        commits_response = requests.get(commits_url, headers=headers, timeout=30)
                        commits_response.raise_for_status()
                        commits_data = commits_response.json()
                        
                        if commits_data.get('count', 0) > 0:
                            # Get latest commit on this branch
                            latest_commit = commits_data['value'][0]
                            commit_date_str = latest_commit['committer']['date']
                            
                            # Parse date
                            if 'Z' in commit_date_str:
                                commit_date = datetime.strptime(commit_date_str, '%Y-%m-%dT%H:%M:%SZ')
                            else:
                                commit_date = datetime.fromisoformat(commit_date_str.replace('Z', '+00:00'))
                            
                            # Track if this branch is active
                            if commit_date >= threshold_date:
                                active_branches_list.append(f"{branch_name} ({(datetime.now() - commit_date).days}d)")
                            
                            # Update most recent commit across all branches
                            if most_recent_date is None or commit_date > most_recent_date:
                                most_recent_date = commit_date
                                most_recent_commit = latest_commit
                                most_active_branch = branch_name
                            
                            # Collect committers from recent commits
                            for commit in commits_data['value'][:5]:
                                all_recent_committers.add(commit['committer']['name'])
                                all_recent_emails.add(commit['committer']['email'])
                    
                    except Exception as branch_error:
                        continue
            
            except Exception as branches_error:
                print(f"⚠️ Could not fetch branches for {repo_name}: {branches_error}")
                continue
        
        else:
            # Original behavior: check only default branch
            commits_url = f"https://dev.azure.com/{organization}/{project}/_apis/git/repositories/{repo_id}/commits?$top=5&api-version=7.0"
            
            try:
                commits_response = requests.get(commits_url, headers=headers, timeout=30)
                commits_response.raise_for_status()
                commits_data = commits_response.json()
                
                if commits_data.get('count', 0) > 0:
                    most_recent_commit = commits_data['value'][0]
                    commit_date_str = most_recent_commit['committer']['date']
                    
                    if 'Z' in commit_date_str:
                        most_recent_date = datetime.strptime(commit_date_str, '%Y-%m-%dT%H:%M:%SZ')
                    else:
                        most_recent_date = datetime.fromisoformat(commit_date_str.replace('Z', '+00:00'))
                    
                    most_active_branch = repo.get('defaultBranch', 'N/A').replace('refs/heads/', '')
                    
                    for commit in commits_data['value']:
                        all_recent_committers.add(commit['committer']['name'])
                        all_recent_emails.add(commit['committer']['email'])
            
            except Exception as commits_error:
                continue
        
        # 🔴 UPDATED: Only add to inactive list if NO activity across ALL branches
        if most_recent_date is None:
            # Completely empty repo
            inactive_repos.append({
                'Repository_Name': repo_name,
                'Last_Commit_Date': 'Never (Empty)',
                'Days_Inactive': 'N/A',
                'Most_Active_Branch': 'N/A',
                'Active_Branches_Count': 0,
                'Last_Committer_Name': 'N/A',
                'Last_Committer_Email': 'N/A',
                'All_Recent_Committers': 'N/A',
                'Commit_Message': 'N/A',
                'Web_URL': repo['webUrl']
            })
        elif most_recent_date < threshold_date:
            # Inactive across all branches
            days_diff = (datetime.now() - most_recent_date).days
            
            inactive_repos.append({
                'Repository_Name': repo_name,
                'Last_Commit_Date': most_recent_date.strftime('%Y-%m-%d'),
                'Days_Inactive': days_diff,
                'Most_Active_Branch': most_active_branch,
                'Active_Branches_Count': len(active_branches_list),
                'Last_Committer_Name': most_recent_commit['committer']['name'],
                'Last_Committer_Email': most_recent_commit['committer']['email'],
                'All_Recent_Committers': ', '.join(list(all_recent_committers)[:5]),
                'Commit_Message': most_recent_commit['comment'].split('\n')[0][:100],
                'Web_URL': repo['webUrl']
            })
    
    except Exception as e:
        print(f"⚠️ Error checking {repo_name}: {e}")

print(f"✅ Scan complete!\n")
